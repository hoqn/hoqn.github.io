개념 및 동작

자료들의 모든 요소를 앞에서부터 탐색하며 이미 정렬된 부분과 비교하여 알맞은 위치를 찾아 삽입한다.

적용

(C Style)

void InsertionSort(Element E[], int n) {
	for (int xIndex = 0; xIndex < n; xIndex++) {
		Element current = E[xIndex];
		int x = current.key;
		int xLoc = shiftVac(E, xIndex, x);
		E[xLoc] = current;
	}
}

int shiftVac(Element E[], int xindex, int x) {
	int vacant, xLoc;
	vacant = xindex;
	xLoc = 0; // Assume failure
	while(vacant > 0) {
		if(E[vacant - 1].key <= x) {
			xLoc = vacant; // Succeed
			break;
		}
		E[vacant] = E[vacant - 1];
		vacant--; // Keep looking
	}
	return xLoc;
}
